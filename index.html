Quick Start Example

// Simple selection and manipulation
$('#myButton').text('Click me!').click(() => {
    alert('Hello from Qry!');
});

// Class manipulation with prefixes
$('.card').cls('+highlight -hidden ~active');

// Create and append elements
const newDiv = $.create('div', {
    text: 'Hello World',
    class: 'card highlight'
});
$('#container').append(newDiv);
